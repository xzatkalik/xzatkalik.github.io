  {zada sa heslo}
  $end:COMPUTERNAME === vrati meno pocitaca

***Tip: Ulozenie opravneni do premennej $cred=Get-Credential

***Vytvorenie sesion ktory sa nemaze
  $s = New-PSSession -ComputerName menopc.domena -Credential $cred (v premennej je ulozene pripojenie)
  Enter-PSSession -Session $s (pripojenie na dane sedenie)
// Toto riesenie je rychlajsie ako to nad
  Remove-PSSession -Session $s //odpoji sedenie $s

***Pripojenie na viac PC naraz
  Invoke-Command //paralelny beh
  $s1 = New-PSSession -ComputerName menopc.domena -Credential $cred 
  $s2 = New-PSSession -ComputerName menopc.domena -Credential $cred 
  Invoke-Command -Session $s1, $s2 -ScriptBlock { $pid } //vypise vysledky podseba
