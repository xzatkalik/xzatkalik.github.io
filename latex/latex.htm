Cislovanie tabuliek, obrazkov, vzorcov

\usepackage{chngcntr}    % - balicek ktory nastavuje cislovania objektov
\counterwithin{objekt}{typ_urovne}    % - zmeni cislovania objektu podla typu_urovne (section, chapter, subsection, susussubsection ...) Priklad prvy obrazok v sekcii 3 bude mat cislo 31
\renewcommand{\theOBJEKT}{\arabic{typ_urovne}.\arabic{objekt}} %    - ovlyvnuje nastavenie cize v tomto pripade prida bodku medzi cilo urovne a cislo obketu. Priklad prvy obrazok v sekcii 3 bude mat cislo 3.1

Zmena cislovania obrazkov

\usepackage{chngcntr}

%\counterwithout{figure}{section}      - neviem naco je zatial
\counterwithin{figure}{section}
\renewcommand{\thefigure}{\arabic{section}.\arabic{figure}}

Zmena cislovania rovnic

\usepackage{chngcntr}

\counterwithin{equation}{section}
\renewcommand{\theequation}{\arabic{section}.\arabic{equation}}

Zmena cislovania Tabuliek

\usepackage{chngcntr}

\counterwithin{table}{section}
\renewcommand{\thetable}{\arabic{section}.\arabic{table}}
